[docs]def factorial(n):
    """Compute Factorial

    Calculates factorial of n.

    :param n: Number to compute factorial of
    :returns: Value of factorial evaluated

    Examples

    >>> from test import mylib
    >>> mylib.factorial(5)
    120
    """

    if n == 0 or n == 1:
        return 1
    return n  * factorial(n-1)
